Validar si puedo convertir un string en float vvv 

entrada = "314.159"

try:
    valor_float = float(entrada)
    print(f"El valor '{entrada}' se puede convertir a float: {valor_float}")
except ValueError:
    print(f"El valor '{entrada}' no se puede convertir a float.")
